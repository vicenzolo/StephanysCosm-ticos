    function calcularMediaEstrelas(produtoNome) {
      const avaliacoes = JSON.parse(localStorage.getItem('avaliacoesProdutos')) || {};
      const produtoAvaliacoes = avaliacoes[produtoNome] || [];

      if (produtoAvaliacoes.length === 0) return { media: 5, total: 0 };

      const soma = produtoAvaliacoes.reduce((acc, av) => acc + av.estrelas, 0);
      const media = soma / produtoAvaliacoes.length;

      return {
        media: Math.round(media * 10) / 10,
        total: produtoAvaliacoes.length
      };
    }

    function gerarEstrelasVisuais(media) {
      const estrelasCheias = Math.floor(media);
      const meiaEstrela = media % 1 >= 0.5;
      const estrelasVazias = 5 - estrelasCheias - (meiaEstrela ? 1 : 0);

      let estrelasHTML = '';

      for (let i = 0; i < estrelasCheias; i++) {
        estrelasHTML += '★';
      }

      if (meiaEstrela) {
        estrelasHTML += '☆';
      }

      for (let i = 0; i < estrelasVazias; i++) {
        estrelasHTML += '☆';
      }

      return estrelasHTML;
    }

      }

      if (produto) {
        const itemExistente = carrinho.find(item => item.nome === produtoNome);
        if (itemExistente) {
          itemExistente.quantidade = (itemExistente.quantidade || 1) + 1;
        } else {
          carrinho.push({ ...produto, preco: parsePreco(produto.preco), quantidade: 1 });
        }

        carrinho = carrinho.map(item => ({ ...item, preco: parsePreco(item.preco) }));
        localStorage.setItem('carrinho', JSON.stringify(carrinho));
        sessionStorage.setItem('carrinhoTemp', JSON.stringify(carrinho));
        mostrarNotificacaoProduto(produto.imagem, produto.nome);
      }
    }

    // Função para remover dos favoritos
    function removerDosFavoritos(produtoNome) {
      favoritos = favoritos.filter(item => item !== produtoNome);
      localStorage.setItem('favoritos', JSON.stringify(favoritos));
      renderizarFavoritos();
      mostrarNotificacao('Removido dos favoritos');
    }

    function toggleFavorito(produtoNome, event) {
      event.preventDefault();
      event.stopPropagation();
      const produto = produtos.find(p => p.nome === produtoNome);
      const imagemSrc = produto ? produto.imagem : '';
      const index = favoritos.indexOf(produtoNome);

      if (index > -1) {
        favoritos.splice(index, 1);
        mostrarNotificacaoProduto(imagemSrc, produtoNome, 'Removido dos favoritos');
      } else {
        favoritos.push(produtoNome);
        mostrarNotificacaoProduto(imagemSrc, produtoNome, 'Adicionado aos favoritos!');
      }

      localStorage.setItem('favoritos', JSON.stringify(favoritos));
      renderizarFavoritos();
    }

    function atualizarDetalhesProduto() {
      const quantidadeEl = document.getElementById('quantidadeProduto');
      const totalEl = document.getElementById('valorTotalProduto');
      if (quantidadeEl) quantidadeEl.value = detalheQuantidade;
      if (totalEl) totalEl.textContent = `R$ ${(detalhePrecoUnitario * detalheQuantidade).toFixed(2).replace('.', ',')}`;
    }

    function ajustarQuantidade(delta) {
      detalheQuantidade = Math.max(1, detalheQuantidade + delta);
      atualizarDetalhesProduto();
    }

    function adicionarAoCarrinhoDetalhe(btn) {
      const painel = document.getElementById('painelDetalhesProduto');
      const nome = painel.querySelector('.detalhes-nome') ? painel.querySelector('.detalhes-nome').textContent : '';
      const produto = produtos.find(p => p.nome === nome);
      if (!produto) return;

      let carrinho = JSON.parse(localStorage.getItem('carrinho')) || [];
      const itemExistente = carrinho.find(item => item.nome === nome);
      if (itemExistente) {
        itemExistente.quantidade = (itemExistente.quantidade || 1) + detalheQuantidade;
      } else {
        carrinho.push({ ...produto, preco: parsePreco(produto.preco), quantidade: detalheQuantidade });
      }
      carrinho = carrinho.map(item => ({ ...item, preco: parsePreco(item.preco) }));
      localStorage.setItem('carrinho', JSON.stringify(carrinho));
      sessionStorage.setItem('carrinhoTemp', JSON.stringify(carrinho));
      mostrarNotificacaoProduto(produto.imagem, produto.nome);
      fecharDetalhesProduto();
    }

    function mostrarDetalhesProduto(produtoNome) {
      const produto = produtos.find(p => p.nome === produtoNome);
      if (!produto) return;

      const painel = document.getElementById('painelDetalhesProduto');
      const { media, total } = calcularMediaEstrelas(produto.nome);
      const estrelasHTML = gerarEstrelasVisuais(media);

      detalheQuantidade = 1;
      detalhePrecoUnitario = parsePreco(produto.preco);

      painel.querySelector('.detalhes-imagem img').src = produto.imagem;
      painel.querySelector('.detalhes-imagem img').alt = produto.alt;
      painel.querySelector('.detalhes-nome').textContent = produto.nome;
      painel.querySelector('.detalhes-tags').innerHTML = produto.categorias.map(cat => `<span class="categoria-tag">${cat}</span>`).join('');
      painel.querySelector('.detalhes-descricao').textContent = produto.descricao;
      painel.querySelector('.detalhes-valor').textContent = `R$ ${(detalhePrecoUnitario * detalheQuantidade).toFixed(2).replace('.', ',')}`;

      // Adicionar estrelas ao painel
      let estrelasHTML_painel = `<div class="stars" style="margin: 10px 0;"><span>${estrelasHTML}</span><span class="stars-count">(${total})</span></div>`;
      const descricao = painel.querySelector('.detalhes-descricao');
      const estrelasExistentes = painel.querySelector('.detalhes-descricao').previousElementSibling;
      if (estrelasExistentes && estrelasExistentes.classList.contains('stars')) {
        estrelasExistentes.remove();
      }
      descricao.insertAdjacentHTML('beforebegin', estrelasHTML_painel);

      atualizarDetalhesProduto();
      painel.classList.add('ativo');
    }

    function fecharDetalhesProduto() {
      const painel = document.getElementById('painelDetalhesProduto');
      if (painel) painel.classList.remove('ativo');
    }

    // Função para mostrar notificações
    function mostrarNotificacao(mensagem) {
      // Criar elemento de notificação
      const notificacao = document.createElement('div');
      notificacao.style.cssText = `
        position: fixed;
        top: 20px;
        right: 20px;
        background: linear-gradient(135deg, #10b981, #059669);
        color: white;
        padding: 12px 20px;
        border-radius: 25px;
        box-shadow: 0 4px 12px rgba(16, 185, 129, 0.3);
        z-index: 1000;
        font-weight: 600;
        animation: slideIn 0.3s ease;
      `;
      notificacao.textContent = mensagem;

      document.body.appendChild(notificacao);

      // Animação de entrada
      const style = document.createElement('style');
      style.textContent = `
        @keyframes slideIn {
          from { transform: translateX(100%); opacity: 0; }
          to { transform: translateX(0); opacity: 1; }
        }
      `;
      document.head.appendChild(style);

      // Remover após 3 segundos
      setTimeout(() => {
        notificacao.style.animation = 'slideOut 0.3s ease';
        setTimeout(() => {
          document.body.removeChild(notificacao);
        }, 300);
      }, 3000);

      // Animação de saída
      style.textContent += `
        @keyframes slideOut {
          from { transform: translateX(0); opacity: 1; }
          to { transform: translateX(100%); opacity: 0; }
        }
      `;
    }

    function mostrarNotificacaoProduto(imagem, nome, mensagem = 'Adicionado ao carrinho') {
      const container = document.getElementById('notificacaoContainer');
      if (!container) {
        mostrarNotificacao(mensagem);
        return;
      }

      const notificacao = document.createElement('div');
      notificacao.className = 'notificacao notificacao-produto';
      notificacao.innerHTML = `
        <div class="notificacao-produto-conteudo">
          <img src="${imagem}" alt="${nome}" class="notificacao-produto-img">
          <div class="notificacao-produto-info">
            <div class="notificacao-produto-nome">${nome}</div>
            <div class="notificacao-produto-mensagem">${mensagem}</div>
          </div>
        </div>
        <div class="notificacao-barra"></div>
      `;
      container.appendChild(notificacao);
      setTimeout(() => notificacao.classList.add('ativo'), 10);

      setTimeout(() => {
        notificacao.classList.remove('ativo');
        setTimeout(() => {
          if (notificacao.parentNode) {
            notificacao.parentNode.removeChild(notificacao);
          }
        }, 300);
      }, 3000);
    }

    function criarConfetes(btn) {
      const rect = btn.getBoundingClientRect();
      const x = rect.left + rect.width / 2;
      const y = rect.top + rect.height / 2;
      const cores = ['#e11d48', '#fb7185', '#fda4af', '#fecdd3', '#f87171'];

      for (let i = 0; i < 15; i++) {
        const confete = document.createElement('div');
        confete.classList.add('confete');

        const cor = cores[Math.floor(Math.random() * cores.length)];
        const tamanho = Math.random() * 8 + 4;
        const angulo = (Math.PI * 2 * i) / 15 + (Math.random() - 0.5);
        const velocidade = Math.random() * 200 + 150;

        const tx = Math.cos(angulo) * velocidade;
        const ty = Math.sin(angulo) * velocidade - 100;

        confete.style.left = x + 'px';
        confete.style.top = y + 'px';
        confete.style.width = tamanho + 'px';
        confete.style.height = tamanho + 'px';
        confete.style.backgroundColor = cor;
        confete.style.setProperty('--tx', tx + 'px');
        confete.style.setProperty('--ty', ty + 'px');
        confete.style.animation = 'confeteExplodir 1.5s ease-out forwards';

        document.body.appendChild(confete);
        setTimeout(() => confete.remove(), 1500);
      }
    }

    function enviarMensagemChat() {
      const input = document.getElementById('chatbotInput');
      const mensagem = input.value.trim();
      if (mensagem === '') return;
      adicionarMensagem(mensagem, 'usuario');
      input.value = '';
      setTimeout(() => {
        const resposta = gerarRespostaAssistente(mensagem.toLowerCase());
        adicionarMensagem(resposta, 'assistente');
      }, 800);
    }
