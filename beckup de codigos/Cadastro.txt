    function voltarParaPrincipal() {
      window.location.href = 'index.html';
    }

    function mostrarAba(aba) {
      document.querySelectorAll('.cadastro-tab').forEach(tab => {
        tab.classList.toggle('ativo', tab.getAttribute('data-aba') === aba);
      });
      
      document.querySelectorAll('.area').forEach(area => {
        area.classList.toggle('ativo', area.id === aba);
      });
    }

    function formatarNome(input) {
      let valor = input.value;
      valor = valor.replace(/[^a-záéíóúàâêôãõç\s]/gi, '');
      valor = valor.split(' ').map(palavra => {
        if (palavra.length === 0) return '';
        return palavra.charAt(0).toUpperCase() + palavra.slice(1).toLowerCase();
      }).join(' ');
      input.value = valor;
    }

    function formatarCPF(input) {
      let valor = input.value.replace(/\D/g, '');
      if (valor.length <= 3) {
        input.value = valor;
      } else if (valor.length <= 6) {
        input.value = valor.substring(0, 3) + '.' + valor.substring(3);
      } else if (valor.length <= 9) {
        input.value = valor.substring(0, 3) + '.' + valor.substring(3, 6) + '.' + valor.substring(6);
      } else {
        input.value = valor.substring(0, 3) + '.' + valor.substring(3, 6) + '.' + valor.substring(6, 9) + '-' + valor.substring(9, 11);
      }
    }

    function formatarCEP(input) {
      let valor = input.value.replace(/\D/g, '');
      if (valor.length <= 5) {
        input.value = valor;
      } else {
        input.value = valor.substring(0, 5) + '-' + valor.substring(5, 8);
      }
    }

    function formatarCartao(input) {
      let valor = input.value.replace(/\D/g, '');
      valor = valor.match(/.{1,4}/g)?.join(' ') || valor;
      input.value = valor;
    }

    function formatarValidade(input) {
      let valor = input.value.replace(/\D/g, '');
      if (valor.length >= 2) {
        input.value = valor.substring(0, 2) + '/' + valor.substring(2, 4);
      } else {
        input.value = valor;
      }
    }

    function validarApenasNumeros(input) {
      input.value = input.value.replace(/\D/g, '');
    }

    function validarApenasLetras(input) {
      input.value = input.value.replace(/[^a-záéíóúàâêôãõç\s]/gi, '');
    }

    function toggleSenha() {
      const input = document.getElementById('senha');
      const btn = event.currentTarget;
      
      if (input.type === 'password') {
        input.type = 'text';
      } else {
        input.type = 'password';
      }
    }

    function validarSenha() {
      const senha = document.getElementById('senha').value;
      
      const comprimento = senha.length >= 8;
      document.getElementById('regraComprimento').classList.toggle('valida', comprimento);
      document.getElementById('regraComprimento').querySelector('.icone-regra').textContent = comprimento ? '✓' : '✗';
      
      const maiuscula = /[A-Z]/.test(senha);
      document.getElementById('regraMaiuscula').classList.toggle('valida', maiuscula);
      document.getElementById('regraMaiuscula').querySelector('.icone-regra').textContent = maiuscula ? '✓' : '✗';
      
      const minuscula = /[a-z]/.test(senha);
      document.getElementById('regraMinuscula').classList.toggle('valida', minuscula);
      document.getElementById('regraMinuscula').querySelector('.icone-regra').textContent = minuscula ? '✓' : '✗';
      
      const numero = /\d/.test(senha);
      document.getElementById('regraNumero').classList.toggle('valida', numero);
      document.getElementById('regraNumero').querySelector('.icone-regra').textContent = numero ? '✓' : '✗';
      
      const simbolo = /[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]/.test(senha);
      document.getElementById('regraSimbolo').classList.toggle('valida', simbolo);
      document.getElementById('regraSimbolo').querySelector('.icone-regra').textContent = simbolo ? '✓' : '✗';
    }

    function validarDados() {
      const nome = document.getElementById('nome').value.trim();
      const sobrenome = document.getElementById('sobrenome').value.trim();
      const cpf = document.getElementById('cpf').value.trim();
      const email = document.getElementById('email').value.trim();
      const senha = document.getElementById('senha').value.trim();

      if (!nome || !sobrenome || !cpf || !email || !senha) {
        mostrarNotificacao('Por favor, preencha todos os campos!');
        return;
      }

      const regexEmail = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!regexEmail.test(email)) {
        mostrarNotificacao('E-mail inválido!');
        return;
      }

      if (senha.length < 8 || !/[A-Z]/.test(senha) || !/[a-z]/.test(senha) || !/\d/.test(senha) || !/[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]/.test(senha)) {
        mostrarNotificacao('A senha não atende aos requisitos!');
        return;
      }

      mostrarAba('enderecos');
    }

    function adicionarEndereco() {
      const container = document.getElementById('containerEnderecos');
      const quantidade = container.querySelectorAll('.endereco-item').length;

    function validarEnderecos() {
      const enderecos = document.querySelectorAll('.endereco-item');
      let temValido = false;

      enderecos.forEach(item => {
        const nome = item.querySelector('.endereco-nome').value.trim();
        const rua = item.querySelector('.endereco-rua').value.trim();
        const cep = item.querySelector('.endereco-cep').value.trim();
        
        if (nome && rua && cep) temValido = true;
      });

      if (!temValido) {
        mostrarNotificacao('Adicione pelo menos um endereço válido!');
        return;
      }

      mostrarAba('pagamento');
    }

    function togglePagamento(tipo) {
      const btn = document.getElementById('btn' + tipo.charAt(0).toUpperCase() + tipo.slice(1));
      const campos = document.getElementById('campos' + tipo.charAt(0).toUpperCase() + tipo.slice(1));
      
      btn.classList.toggle('ativo');
      campos.classList.toggle('ativo');
    }

    function criarConta() {
      const nome = document.getElementById('nome').value.trim();
      const sobrenome = document.getElementById('sobrenome').value.trim();
      const cpf = document.getElementById('cpf').value.trim();
      const email = document.getElementById('email').value.trim();
      const senha = document.getElementById('senha').value.trim();

      // Coletar endereços
      const enderecos = [];
      document.querySelectorAll('.endereco-item').forEach(item => {
        const nomeEnd = item.querySelector('.endereco-nome').value.trim();
        const rua = item.querySelector('.endereco-rua').value.trim();
        const cep = item.querySelector('.endereco-cep').value.trim();
        
        if (nomeEnd && rua && cep) {
          enderecos.push({ nome: nomeEnd, rua, cep });
        }
      });

      if (enderecos.length === 0) {
        mostrarNotificacao('Adicione pelo menos um endereço!');
        return;
      }

      // Coletar métodos de pagamento
      const metodos = [];
      
      if (document.getElementById('btnCredito').classList.contains('ativo')) {
        const numero = document.getElementById('numeroCredito').value.trim();
        const nome = document.getElementById('nomeCredito').value.trim();
        const validade = document.getElementById('validadeCredito').value.trim();
        const cvv = document.getElementById('cvvCredito').value.trim();
        if (numero && nome && validade && cvv) {
          metodos.push({ tipo: 'credito', numero, nome, validade, cvv });
        }
      }

      if (document.getElementById('btnDebito').classList.contains('ativo')) {
        const numero = document.getElementById('numeroDebito').value.trim();
        const nome = document.getElementById('nomeDebito').value.trim();
        const validade = document.getElementById('validadeDebito').value.trim();
        const cvv = document.getElementById('cvvDebito').value.trim();
        if (numero && nome && validade && cvv) {
          metodos.push({ tipo: 'debito', numero, nome, validade, cvv });
        }
      }

      if (document.getElementById('btnPix').classList.contains('ativo')) {
        const chave = document.getElementById('chavePix').value.trim();
        if (chave) {
          metodos.push({ tipo: 'pix', chave });
        }
      }

      if (metodos.length === 0) {
        mostrarNotificacao('Adicione pelo menos um método de pagamento!');
        return;
      }

      // Salvar na janela principal (parent)
      try {
        const usuario = {
          nome,
          sobrenome,
          cpf,
          email,
          senha,
          enderecos,
          metodosPagamento: metodos,
          dataCriacao: new Date().toISOString()
        };

        // Salvar no localStorage
        localStorage.setItem(`usuario_${cpf}`, JSON.stringify(usuario));
        localStorage.setItem('usuarioLogado', JSON.stringify(usuario));

        mostrarNotificacao('Conta criada com sucesso!');
        
        setTimeout(() => {
          window.location.href = 'index.html';
        }, 1500);
      } catch (erro) {
        mostrarNotificacao('Erro ao criar conta!');
        console.error(erro);
      }
    }

    function mostrarNotificacao(msg) {
      const div = document.createElement('div');
      div.className = 'notification';
      div.textContent = msg;
      document.body.appendChild(div);
      
      setTimeout(() => {
        div.remove();
      }, 3000);
    }

    function esconderSpinnerDeCarregamento() {
      const spinner = document.getElementById('pageLoadSpinner');
      if (!spinner) return;
      spinner.classList.add('hidden');
    }

    window.addEventListener('load', function() {
      setTimeout(() => {
        esconderSpinnerDeCarregamento();
      }, 2500);
    });
